# Clustering into relevant cluster of low medium or high values
for i in df.columns:
    if i == 'id' or i == 'class':
        continue
    if i == 'clump_thickness' or i == 'normal_nucleoli' :
        # 3 clusters for clump_thickness and normal_nucleoli
        kmeans = KMeans(n_clusters=3)
    else:
        # 2 clusters for the rest
        kmeans = KMeans(n_clusters=2)
    
    X = np.asarray(df[i]).reshape((-1,1))
    y = np.asarray(df['class']).reshape((-1,1))
    kmeans.fit(X,y)
    pred = kmeans.predict(X).reshape((-1,1))
    data = np.hstack((X,pred))
    data = pd.DataFrame(data)
    data.columns = [i,'predClass']
    #data.head()
    sns.countplot(i, data = data, hue='predClass')
    plt.show()
